func ohm(v: Double, i: Double, r: Double){
  if(v == 0 && i == 0 && r == 0){
    print("Invalid values")
  }
  else if (v != 0 && i != 0 && r != 0){
    print("Invalid values")
  }
  else if(v == 0 && i != 0 && r != 0){
    print(String(format: "%.2f", i * r))
  }
  else if(i == 0 && v != 0 && r != 0){
    print(String(format: "%.2f", v/r))
  }
  else{
   print(String(format: "%.2f", v/i))
  }
}
